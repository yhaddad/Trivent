### LIBRARY #################################################################

# definitions to pass to the compiler
ADD_DEFINITIONS( "-Wall -ansi -pedantic" )
ADD_DEFINITIONS( "-Wno-long-long" )

# include directories
INCLUDE_DIRECTORIES( ./include ../EventBuilder/include )
INSTALL_DIRECTORY( ./include DESTINATION . FILES_MATCHING PATTERN "*.h" )
INSTALL_DIRECTORY( ./include DESTINATION . FILES_MATCHING PATTERN "*.hh" )

# add library
AUX_SOURCE_DIRECTORY( ./src library_sources )
ADD_SHARED_LIBRARY( ${PROJECT_NAME}Reco ${library_sources} )
INSTALL_SHARED_LIBRARY( ${PROJECT_NAME}Reco DESTINATION lib )
