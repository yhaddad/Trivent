63 void SDHCAL_RawBuffer_Navigator::setSCBuffer()
64 {
65  if (! hasSlowControlData() ) return;
66  if (0 != _SCbuffer.getsize() ) return; //deja fait
67  if (_badSCdata) return;
68  _SCbuffer.first=&(getDIFBufferStart()[getEndOfDIFData()]);
69  //compute Slow Control size
70  uint32_t maxsize=_buffer.getsize()-_DIFstartIndex-getEndOfDIFData()+1; // should I +1 here ?
71  uint32_t k=1; //SC Header
72  uint32_t dif_ID=_SCbuffer.first[1];
73  uint32_t chipSize=_SCbuffer.first[3];
74  while ( (dif_ID != 0xa1 && _SCbuffer.first[k] != 0xa1 && k <maxsize) ||
75  (dif_ID == 0xa1 && _SCbuffer.first[k+2]==chipSize && k<maxsize) )
76  {
77  k+=2; //DIF ID + ASIC Header
78  uint32_t scsize=_SCbuffer.first[k];
79  if (scsize != 74 && scsize != 109)
80  {
81  std::cout << "PROBLEM WITH SC SIZE " << scsize << std::endl;
82  k=0;
83  _badSCdata=true;
84  break;
85  }
86  k++; //skip size bit
87  k+=scsize; // skip the data
88  }
89  if (_SCbuffer.first[k] == 0xa1 && !_badSCdata ) _SCbuffer.second=k+1; //add the trailer
90  else
91  {
92  _badSCdata=true;
93  std::cout << "PROBLEM SC TRAILER NOT FOUND " << std::endl;
94  }
95 }
